#include <iostream>
using namespace std;

int BinarySearch(int arr[],int n, int key){
     int s = 0;
     int e = n-1;
     int mid = (s+e)/2;
     while (s<=n){
          if(arr[mid] == key){
               return mid;
          }else if(arr[mid] > key){
               return e = mid-1;
               
          }else{
               return s = mid+1; 
          }
     }
     return -1 ;
     
}

int main () {
     
     int n;
     cin>>n;
     int arr[n];
     for(int i=0;i<n;i++){
          cin>>arr[i];
     }
     int key;
     cin>>key;
     
     
     cout<<BinarySearch(arr, n , key)<<"\n";
     return 0;
}
